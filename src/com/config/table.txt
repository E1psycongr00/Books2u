-- 책 테이블
DROP TABLE book CASCADE CONSTRAINTS;
create table book
(
  isbn varchar2(20) primary key,
  title varchar2(500) not null,
  author varchar2(500) not null,
  publisher varchar2(500) not null,
  publishDate date not null,
  story varchar2(2500) not null,
  genre varchar2(500) not null,
  price number(10) not null
);


-- 회원 테이블
DROP TABLE member CASCADE CONSTRAINTS;
create table member
(
  userId varchar2(10) primary key,
  passWd varchar2(10) not null,
  userName varchar2(10) not null,
  ssn1 varchar2(10) not null,
  ssn2 varchar2(10) not null,
  phone1 varchar2(3) not null,
  phone2 varchar2(4) not null,
  phone3 varchar2(4) not null,
  post varchar2(6) not null,
  addr1 varchar2(20) not null,
  addr2 varchar2(30) not null,
  addr3 varchar2(30),
  email1 varchar2(20) not null,
  email2 varchar2(20) not null,
  joinDate date not null,
  lastDate date not null
);


-- 리뷰 테이블
DROP TABLE review CASCADE CONSTRAINTS;
create table review
(
  reviewId number(10),
  userId varchar2(10) not null,
  rating number(2, 1) not null,
  content varchar2(10) not null,
  reviewDate date not null,
  isbn varchar2(10) not null,

  constraint review_isbn_fk foreign key(isbn) references book(isbn),
  constraint reviewid_pk primary key(reviewId)
);


-- 주문 테이블
DROP TABLE ordered CASCADE CONSTRAINTS;
create table ordered
(
  orderId number(20) not null,
  isbn varchar2(10) not null,
  userId varchar2(10) not null,
  amount number(10) not null,
  recipient varchar2(10) not null,
  recipientPost varchar2(10) not null,
  recipientAddr1 varchar2(20) not null,
  recipientAddr2 varchar2(30) not null,
  recipientAddr3 varchar2(30),
  receiptPhone1 varchar2(3) not null,
  receiptPhone2 varchar2(4) not null,
  receiptPhone3 varchar2(4) not null,
  payment varchar2(10) not null,
  orderDate date not null,

  constraint ordered_isbn_fk foreign key(isbn) references book(isbn),
  constraint orderid_pk primary key(orderId)
);
